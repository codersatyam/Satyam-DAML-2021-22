1. git --version ## (this will give version of git)
2. git --help    ## (it will show you all commands)
3. git clone url ## (for clonning a repo )
4. git add .     ## (to add all the changes to the repo)
5. git commit -m " " ## (to add comment to the changes files)
6. git status        ## (to check the status of repo)
7. git push origin main ## (to push your code into github)
8. git branch           ##( tio chekc all the branches ) 
9. git branch [branch name]   ##( create a new branch at the current commit)
10. git log                   ##(show all commits in the current branch’s history)  
11. git checkout branch name  ## (switch to another branch and check it out into your working directory)
12. git merge branch name                ## (merge the specified branch’s history into the current one)
